284        self.thread_pool.setMaxThreadCount(self.max_threads)
285        self.operation_pool: list[Operation] = []

288        self.abort_all_on_error = abort_all_on_error

292    def add(self, operation: Operation):
293        self.operation_pool.append(operation)
294
295    def start(self):
296        self.interaction.on_runner_start()
297        self.progress_obj = QueueProgress(QueueProgressMode.SINGLE, len(self.operation_pool))
298        for op in self.operation_pool:
299            self.progress_obj.add_single(op.id)
300        for op in self.operation_pool:
301            self.__start_next_operation()
302
303    def stop(self):
304        self.interaction.on_runner_stop()
305        self.thread_pool.waitForDone()
306        self.active_operations = 0
307        self.operation_pool.clear()
308
309    def __start_next_operation(self):
310        can_start = self.active_operations < self.thread_pool.maxThreadCount()
311        if can_start and self.operation_pool:
312            op = self.operation_pool.pop(0)
313            op.set_finished_callback(lambda: self.on_operation_finished(op))
314            op.set_op_progress_callback(self.on_op_progress_update)
315            self.thread_pool.start(op)
316            self.active_operations += 1
317
318    def on_operation_finished(self, operation: Operation):
319        if self.abort_all_on_error and operation.is_failed():
320            logger.error("Operation %s failed, stopping all operations", operation.id)
321            self.__set_runner_finished()
322            return
323        self.active_operations -= 1
324        self.__start_next_operation()
325        if self.active_operations == 0 and not self.operation_pool:
326            self.__set_runner_finished()
327
328    def __set_runner_finished(self):
329        time.sleep(1)
330        statistics = RunnerStatistics(self.operation_pool)
331        self.interaction.on_runner_finish(statistics)
332        self.on_runner_finished() if self.on_runner_finished else None
333
334    def on_op_progress_update(self, op_id: str, op_progress: int):
335        self.progress_obj.set_single_progress(op_id, op_progress)
336        self.interaction.on_runner_update_progress(self.progress_obj.get_total_progress())

Base class for protocol classes.

Protocol classes are defined as::

class Proto(Protocol):
    def meth(self) -> int:
        ...

Such classes are primarily used with static type checkers that recognize structural subtyping (static duck-typing).

For example::

class C:
    def meth(self) -> int:
        return 0

def func(x: Proto) -> int:
    return x.meth()

func(C())  # Passes static type check

See PEP 544 for details. Protocol classes decorated with @typing.runtime_checkable act as simple-minded runtime protocols that check only the presence of given attributes, ignoring their type signatures. Protocol classes can be generic, they are defined as::

class GenProto[T](Protocol):
    def meth(self) -> T:
        ...

1717def _no_init_or_replace_init(self, *args, **kwargs):
1718    cls = type(self)
1719
1720    if cls._is_protocol:
1721        raise TypeError('Protocols cannot be instantiated')
1722
1723    # Already using a custom `__init__`. No need to calculate correct
1724    # `__init__` to call. This can lead to RecursionError. See bpo-45121.
1725    if cls.__init__ is not _no_init_or_replace_init:
1726        return
1727
1728    # Initially, `__init__` of a protocol subclass is set to `_no_init_or_replace_init`.
1729    # The first instantiation of the subclass will call `_no_init_or_replace_init` which
1730    # searches for a proper new `__init__` in the MRO. The new `__init__`
1731    # replaces the subclass' old `__init__` (ie `_no_init_or_replace_init`). Subsequent
1732    # instantiation of the protocol subclass will thus use the new
1733    # `__init__` and no longer call `_no_init_or_replace_init`.
1734    for base in cls.__mro__:
1735        init = base.__dict__.get('__init__', _no_init_or_replace_init)
1736        if init is not _no_init_or_replace_init:
1737            cls.__init__ = init
1738            break
1739    else:
1740        # should not happen
1741        cls.__init__ = object.__init__
1742
